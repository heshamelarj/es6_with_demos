FACTORY PATTERN

- Factory Pattern is used when we have multiple objects share some same attributs or methods

function MemberFactory(){
this.createMember = function(name,type){
let memeber;
switch (type)
{
case "simple" :
memeber = new SimpleMembership(name);
case "standard" :
memeber = new StandardMembership(name);
case "supper" :
memeber = new SupperMembership(name);
}
memeber.type = type;
memeber.define = function(){
console.log(`${this.name} (${this.type}) ${this.cost}`);

}
return memeber;
}
}
